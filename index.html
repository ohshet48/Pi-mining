async function login() { const email = document.getElementById('loginEmail').value; const password = document.getElementById('loginPassword').value; if (!email || !password) { showError('Please enter email and password'); return; } try { // Firebase Authentication ile giriş const userCredential = await firebase.auth().signInWithEmailAndPassword(email, password); const user = userCredential.user; // Backend'den kullanıcı bilgilerini al const response = await fetch(API_ENDPOINTS.getUserData + '?userId=' + user.uid); const data = await response.json(); if (response.ok) { localStorage.setItem('userId', user.uid); localStorage.setItem('userEmail', user.email); localStorage.setItem('isLoggedIn', 'true'); showSuccess('Login successful!'); setTimeout(() => { document.getElementById('loginSection').style.display = 'none'; document.getElementById('mineSection').style.display = 'block'; }, 1000); } else { showError(data.error || 'Login failed'); } } catch (error) { console.error('Login error:', error); showError('Invalid email or password'); } }
